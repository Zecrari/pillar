Creating a Router #

A Router collects route definitions for one domain. Register it with the app using include_router().

domains/products/router.pypython
from pillar import Router

router = Router(
    prefix="/products",  # prepended to every route in this router
    tags=["Products"],   # grouped in Swagger UI
)
main.pypython
from pillar import Pillar
from domains.products.router import router as products_router
from domains.users.router    import router as users_router

app = Pillar(title="My API")
app.include_router(products_router)
app.include_router(users_router)

HTTP Methods #

pythonpython
@router.get("/")           # list / read
@router.post("/")          # create  (default status_code=201)
@router.put("/{id}")       # full update
@router.patch("/{id}")     # partial update
@router.delete("/{id}")    # remove  (return 204 + None)
@router.head("/")          # headers only (no body)
@router.options("/")       # CORS preflight

Path Parameters #

Use {name} in the path. Pillar coerces the string to the annotated type automatically.

pythonpython
@router.get("/{product_id}/reviews/{review_id}")
async def get_review(product_id: int, review_id: int):
    # Both coerced to int — no manual int() call needed
    return {"product": product_id, "review": review_id}

@router.get("/{slug}")
async def get_by_slug(slug: str):
    return {"slug": slug}

Query Parameters #

Parameters not in the path and not Pydantic models are treated as query params. Defaults make them optional.

pythonpython
@router.get("/search")
async def search(

    limit:   int   = 20,
    offset:  int   = 0,
    sort:    str   = "created_at",
    order:   str   = "desc",
    active:  bool  = True,
):

Raw Request Access #

pythonpython
from starlette.requests import Request

@router.post("/webhook")
async def webhook(request: Request):
    body     = await request.body()          # raw bytes
    payload  = await request.json()          # parsed JSON
    token    = request.headers.get("X-Signature", "")
    ip       = request.client.host
    qs       = dict(request.query_params)
    return {"received": len(body), "ip": ip}

Custom Responses #

pythonpython
from starlette.responses import (
    JSONResponse, PlainTextResponse,
    HTMLResponse, Response, RedirectResponse,
)

@router.get("/csv-export")
async def csv(service: ReportService):
    data = service.generate_csv()
    return PlainTextResponse(data, media_type="text/csv",
                             headers={"Content-Disposition": 'attachment; filename="report.csv"'})

@router.get("/binary/{file_id}")
async def download(file_id: str, store: FileStore):
    content = store.get(file_id)
    return Response(content, media_type="application/octet-stream")

@router.get("/old-url")
async def redirect():
    return RedirectResponse(url="/new-url", status_code=301)

@router.post("/async-job")
async def create_job(data: JobRequest, svc: JobService):
    job_id = svc.enqueue(data)
    return JSONResponse({"job_id": job_id}, status_code=202,
                        headers={"Location": f"/jobs/{job_id}"})

How Dependency Injection Works #

Pillar reads your handler's type hints at call time and resolves each parameter from the DI container — recursively. The entire dependency graph is constructed once and cached as singletons.

automatic resolution chainpython
# Handler needs UserService
# UserService.__init__(self, repo: UserRepository)
# UserRepository.__init__(self, db: Database)
#   → Database is registered as a singleton by Pillar at startup
#   → UserRepository is constructed with that db
#   → UserService is constructed with that repo
#   → All cached for the lifetime of the app

@router.get("/{user_id}")
async def get_user(user_id: int, service: UserService):
    # 'service' is fully wired — zero configuration required
    return service.get_user(user_id)

Injection Precedence #

PriorityConditionResult
1Parameter name matches {path_param}Coerced to annotated type
2Annotation is a Pydantic BaseModelParsed from JSON body
3Annotation is a non-primitive classResolved from DI container
4Annotation is RequestRaw Starlette request object
5Everything else (str, int, etc.)Query parameter (with optional default)

    """Start every test with a clean DI container."""

Pydantic Request Bodies #

Type-annotate a handler parameter with any Pydantic BaseModel and Pillar parses the JSON body into it automatically.

pythonpython
from pydantic import BaseModel
from typing import Optional

class CreateOrderRequest(BaseModel):
    product_id: int
    quantity:   int
    notes:      Optional[str] = None

class OrderResponse(BaseModel):
    id:         int
    product_id: int
    quantity:   int
    total:      float
    status:     str

@router.post("/", status_code=201, response_model=OrderResponse)
async def create_order(data: CreateOrderRequest, svc: OrderService):
    return svc.create(data.product_id, data.quantity, data.notes)

Field Validation #

pythonpython
from pydantic import BaseModel, Field, field_validator
from typing import Annotated

Price = Annotated[float, Field(gt=0, description="Must be positive")]

class ProductCreate(BaseModel):
    name:        str   = Field(min_length=1, max_length=200)
    price:       Price
    category:    str   = Field(pattern=r"^[a-z_]+$")
    stock:       int   = Field(ge=0, default=0)
    description: str   = Field(default="", max_length=2000)

    @field_validator("name")
    @classmethod
    def name_not_blank(cls, v: str) -> str:
        if not v.strip():
            raise ValueError("name must not be blank")
        return v.strip()

Raising Errors #

Raise any Pillar exception from any layer — Pillar catches it and returns the right HTTP response automatically. No try/except in handlers needed.

pythonpython
from pillar.exceptions import (
    NotFoundError,      # 404
    UnauthorizedError,  # 401
    ForbiddenError,     # 403
    ConflictError,      # 409
    ValidationError,    # 422
)

class ProductService:
    def get(self, product_id: int) -> dict:
        product = self.repo.get(product_id)
        if not product:
            raise NotFoundError(f"Product {product_id} not found")
        return product

    def create(self, data: dict, actor: dict) -> dict:
        if actor["role"] not in ("admin", "manager"):
            raise ForbiddenError("Only admins and managers can create products")

        if self.repo.sku_exists(data["sku"]):
            raise ConflictError(f"SKU '{data['sku']}' already exists")

        return self.repo.create(data)

    def publish(self, product_id: int, actor: dict) -> dict:
        if not actor.get("verified"):
            raise UnauthorizedError("Email verification required")
ExceptionStatusDefault detail
NotFoundError404Resource not found
UnauthorizedError401Authentication required
ForbiddenError403Permission denied
ConflictError409Resource already exists
ValidationError422Validation failed

CORS Configuration #

pillar.toml — production CORStoml
[cors]
enabled           = true
allow_origins     = ["https://myapp.com", "https://staging.myapp.com"]
allow_methods     = ["GET", "POST", "PUT", "PATCH", "DELETE"]
allow_headers     = ["Authorization", "Content-Type", "X-Request-ID"]
allow_credentials = true
expose_headers    = ["X-Request-ID", "X-Response-Time"]
💡 Wildcard origins

Setting allow_origins = ["*"] disables cookie/credential sharing. If you need allow_credentials = true, list each origin explicitly.

Basic WebSocket #

domains/realtime/router.pypython
from pillar import Router
from starlette.websockets import WebSocket, WebSocketDisconnect

router = Router(prefix="/realtime", tags=["Realtime"])

@router.websocket("/ws/ping")
async def ping(websocket: WebSocket):
    await websocket.accept()
    try:
        while True:
            msg = await websocket.receive_text()
            await websocket.send_text(f"pong: {msg}")
    except WebSocketDisconnect:
        pass   # client closed normally

WebSocket with DI #

The same injection rules as HTTP handlers apply — type-hint a parameter and Pillar injects it.

pythonpython
@router.websocket("/ws/notifications/{user_id}")
async def notifications(
    websocket: WebSocket,
    user_id:   int,              # ← path param
    svc:       NotificationService,  # ← DI injected
):
    await websocket.accept()
    async for event in svc.stream_events(user_id):
        await websocket.send_json(event)

Chat Rooms Pattern #

domains/chat/router.pypython
from pillar import Router
from starlette.websockets import WebSocket, WebSocketDisconnect
from collections import defaultdict
from typing import Dict, Set

router = Router(prefix="/chat", tags=["Chat"])

# In-memory room registry (use Redis for multi-process)
_rooms: Dict[str, Set[WebSocket]] = defaultdict(set)

@router.websocket("/ws/{room_id}")
async def chat_room(websocket: WebSocket, room_id: str):
    await websocket.accept()
    _rooms[room_id].add(websocket)
    try:
        while True:
            msg = await websocket.receive_text()
            # Broadcast to everyone in the room
            dead = set()
            for ws in _rooms[room_id]:
                try:
                    await ws.send_text(f"[{room_id}] {msg}")
                except Exception:
                    dead.add(ws)
            _rooms[room_id] -= dead
    except WebSocketDisconnect:
        _rooms[room_id].discard(websocket)
